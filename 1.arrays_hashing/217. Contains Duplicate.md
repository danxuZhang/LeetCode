# 217. Contains Duplicate

Hash Set:

``` c++
class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        std::unordered_set<int> us;

        for (int num : nums) {
            if (us.count(num) != 0) 
                return true;
            us.insert(num);
        }

        return false;
    }
};

```

Double Pointers:

``` c++
class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        sort(nums.begin(), nums.end());

        if (nums.size() < 2)
            return false;

        int slow = 0;
        int fast = 1;

        while (fast < nums.size()) {
            if (nums[fast] == nums[slow])
                return true;
            fast++;
            slow++; 
        }

        return false;
    }
};
```
